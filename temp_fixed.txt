// --- 2. FLUID BACKGROUND ANIMATION ---
const canvas = document.getElementById('fluid-canvas');
const ctx    = canvas.getContext('2d');
let width, height;
let particles = [];
function getParticleCount() {
  const w = window.innerWidth;
  if (w < 480) return 5;
  if (w < 768) return 8;
  if (!isMouseDevice) return 10;
  return 18;
}
let PARTICLE_COUNT = getParticleCount();
const dpr = window.devicePixelRatio || 1;

function resize() {
  const heroEl = document.getElementById('hero');
  const w = heroEl.offsetWidth;
  const h = heroEl.offsetHeight;
  canvas.width  = w * dpr;
  canvas.height = h * dpr;
  canvas.style.width  = w + 'px';
  canvas.style.height = h + 'px';
  ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
  width  = w;
  height = h;
  const newCount = getParticleCount();
  if (newCount !== PARTICLE_COUNT) {
    PARTICLE_COUNT = newCount;
    while (particles.length > PARTICLE_COUNT) particles.pop();
    while (particles.length < PARTICLE_COUNT) particles.push(new Particle());
  }
}
window.addEventListener('resize', resize);
resize();

function getMaxParticleSize() {
  const w = window.innerWidth;
  if (w < 480) return 40;
  if (w < 768) return 60;
  return 80;
}

class Particle {
  constructor() {
    this.x     = Math.random() * width;
    this.y     = Math.random() * height;
    this.vx    = (Math.random() - 0.5) * 0.8;
    this.vy    = (Math.random() - 0.5) * 0.8;
    const maxSize = getMaxParticleSize();
    this.size  = Math.random() * maxSize + (maxSize * 0.5);
    const colors = ['rgba(80,80,80,0.7)', 'rgba(60,60,60,0.6)', 'rgba(100,100,100,0.5)', 'rgba(40,40,40,0.7)'];
    this.color = colors[Math.floor(Math.random() * colors.length)];
  }
  update() {
    this.x += this.vx;
    this.y += this.vy;
    if (this.x < -this.size || this.x > width  + this.size) this.vx *= -1;
    if (this.y < -this.size || this.y > height + this.size) this.vy *= -1;
  }
  draw() {
    ctx.beginPath();
    ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
    ctx.fillStyle = this.color;
    ctx.fill();
  }
}
for (let i = 0; i < PARTICLE_COUNT; i++) particles.push(new Particle());
function animateFluid() {
  ctx.clearRect(0, 0, width, height);
  particles.forEach(p => { p.update(); p.draw(); });
  requestAnimationFrame(animateFluid);
}
animateFluid();

const fsOverlay = document.getElementById('video-fullscreen');

let activeGridItem = null;

function fsFormat(s) {
  if (isNaN(s)) return '0:00';
  const m = Math.floor(s / 60);
  const sec = Math.floor(s % 60);
  return m + ':' + (sec < 10 ? '0' : '') + sec;
}

function fsUpdateUI() {
  const playing = !fsVideo.paused && !fsVideo.ended;
  const iconPlay = fsPlaypause.querySelector('.icon-play');
  const iconPause = fsPlaypause.querySelector('.icon-pause');
  const iconMuted = fsMute.querySelector('.icon-muted');
  const iconUnmuted = fsMute.querySelector('.icon-unmuted');
  iconPlay.style.display = playing ? 'none' : 'block';
  iconPause.style.display = playing ? 'block' : 'none';
  iconMuted.style.display = fsVideo.muted ? 'block' : 'none';
  iconUnmuted.style.display = fsVideo.muted ? 'none' : 'block';
  fsTimeCurrent.textContent = fsFormat(fsVideo.currentTime);
  fsTimeDuration.textContent = fsFormat(fsVideo.duration);
  if (fsVideo.duration) {
    fsProgressFill.style.width = (fsVideo.currentTime / fsVideo.duration * 100) + '%';
  }
}

function openFullscreen(item) {
  const video = item.querySelector('video');
  const titleEl = item.querySelector('.grid-title');
  const catEl = item.querySelector('.grid-cat');
  if (!video || !video.src) return;

  activeGridItem = item;
  fsVideo.src = video.src;
  fsVideo.currentTime = 0;
  fsVideo.muted = false;
  fsVideo.loop = false;
  fsTitle.textContent = titleEl ? titleEl.textContent : '';
  fsCat.textContent = catEl ? catEl.textContent : '';

  fsOverlay.classList.add('active');
  document.body.style.overflow = 'hidden';

  fsVideo.play().then(() => {
    fsUpdateUI();
  }).catch(() => {});

  // Pause the grid video
  video.pause();
  item.classList.add('playing');
}

function closeFullscreen() {
  fsVideo.pause();
  fsVideo.src = '';
  fsOverlay.classList.remove('active');
  document.body.style.overflow = '';

  if (activeGridItem) {
    const gridVideo = activeGridItem.querySelector('video');
    if (gridVideo) {
      gridVideo.muted = true;
      gridVideo.loop = true;
      gridVideo.currentTime = 0;
      gridVideo.play().catch(() => {});
    }
    activeGridItem.classList.remove('playing');
    activeGridItem = null;
  }
}

// Autoplay only visible videos (Intersection Observer)

const videoObserver = new IntersectionObserver((entries) => {
  entries.forEach(entry => {
    const v = entry.target;
    const item = v.closest('.grid-item');
    if (entry.isIntersecting) {
      if (v.paused && item && !item.classList.contains('playing') && !fsOverlay.classList.contains('active')) {
        v.muted = true;
        v.loop = true;
        v.play().catch(() => {});
      }
    } else {
      if (item && !item.classList.contains('playing')) {
        v.pause();
      }
    }
  });
}, { threshold: 0.25 });
allVideos.forEach(v => videoObserver.observe(v));

// Grid video click handlers — open fullscreen on play
document.querySelectorAll('[data-video-player]').forEach(item => {
  const video = item.querySelector('video');
  const playOverlay = item.querySelector('.play-overlay');
  if (!video) return;

  // Hover = mute all others, unmute this instantly
  item.addEventListener('mouseenter', () => {
    // Mute all other videos instantly
    document.querySelectorAll('[data-video-player]').forEach(other => {
      if (other !== item) {
        const v = other.querySelector('video');
        if (v && !v.paused) { v.muted = true; v.volume = 1; }
      }
    });
    if (!fsOverlay.classList.contains('active') && !video.paused) {
      video.muted = false;
      video.volume = 1;
    }
  });
  item.addEventListener('mouseleave', () => {
    if (!fsOverlay.classList.contains('active')) {
      video.muted = true;
      video.volume = 1;
    }
  });

  // Click play overlay → open fullscreen
  if (playOverlay) {
    playOverlay.addEventListener('click', (e) => {
      e.stopPropagation();
      openFullscreen(item);
    });
  }

  // Click anywhere on video area → open fullscreen
  item.addEventListener('click', (e) => {
    if (e.target.closest('.video-controls')) return;
    if (e.target === playOverlay) return;
    openFullscreen(item);
  });

<script>
// --- AUDIO PLAYER ---
(function() {
  const audio = new Audio();
  let currentTrack = null;

  function audioFmt(s) {
    if (isNaN(s)) return '0:00';
    const m = Math.floor(s / 60);
    const sec = Math.floor(s % 60);
    return m + ':' + (sec < 10 ? '0' : '') + sec;
  }

  function pauseIcon(btn) {
    btn.innerHTML = '<svg viewBox="0 0 24 24" fill="white" width="20" height="20"><rect x="6" y="3" width="4" height="18"/><rect x="14" y="3" width="4" height="18"/></svg>';
  }

  function playIcon(btn) {
    btn.innerHTML = '<svg viewBox="0 0 24 24" fill="white" width="20" height="20"><polygon points="6,3 20,12 6,21"/></svg>';
  }

  function stopAll() {
    document.querySelectorAll('.audio-track').forEach(track => {
      track.classList.remove('playing');
      const btn = track.querySelector('.audio-btn-play');
      playIcon(btn);
      const fill = track.querySelector('.audio-track-progress-fill');
      if (fill) fill.style.width = '0%';
      const time = track.querySelector('.audio-current');
      if (time) time.textContent = '0:00';
    });
  }

  document.querySelectorAll('[data-audio-player]').forEach(track => {
    const btn = track.querySelector('.audio-btn-play');
    const progressBar = track.querySelector('.audio-track-progress-bar');
    const progressFill = track.querySelector('.audio-track-progress-fill');
    const timeEl = track.querySelector('.audio-current');

    function updateProgress() {
      if (audio.duration) {
        progressFill.style.width = (audio.currentTime / audio.duration * 100) + '%';
        timeEl.textContent = audioFmt(audio.currentTime);
      }
    }

    btn.addEventListener('click', (e) => {
      e.stopPropagation();
      if (currentTrack === track && !audio.paused) {
        audio.pause();
        playIcon(btn);
        return;
      }
      if (currentTrack === track && audio.paused) {
        audio.play();
        pauseIcon(btn);
        return;
      }
      stopAll();
      const trackAudio = track.querySelector('audio');
      if (trackAudio && trackAudio.src) {
        audio.src = trackAudio.src;
        audio.load();
      }
      currentTrack = track;
      track.classList.add('playing');
      pauseIcon(btn);
      audio.play().catch(() => {
        playIcon(btn);
        track.classList.remove('playing');
        currentTrack = null;
      });
    });

    progressBar.addEventListener('click', (e) => {
      e.stopPropagation();
      if (currentTrack !== track) return;
      const rect = progressBar.getBoundingClientRect();
      const pct = (e.clientX - rect.left) / rect.width;
      if (audio.duration) {
        audio.currentTime = pct * audio.duration;
      }
    });

    audio.addEventListener('timeupdate', () => {
      if (currentTrack) {
        const fill = currentTrack.querySelector('.audio-track-progress-fill');
        const time = currentTrack.querySelector('.audio-current');
        if (fill && audio.duration) fill.style.width = (audio.currentTime / audio.duration * 100) + '%';
        if (time) time.textContent = audioFmt(audio.currentTime);
      }
    });

    audio.addEventListener('ended', () => {
      stopAll();
      currentTrack = null;
    });
  });
})();
